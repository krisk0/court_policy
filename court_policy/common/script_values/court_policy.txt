# return 100 or 50 0, 100 = yes of course, 50 = maybe, 0 = never
court_policy_fm_sex = {
	faith = {
		if = {
			limit = {
				has_doctrine = doctrine_adultery_men_accepted
				has_doctrine = doctrine_adultery_women_accepted
			}
			add = 100
		}
		else_if = {
			limit = {
				OR = {
					has_doctrine = doctrine_adultery_men_accepted
					has_doctrine = doctrine_adultery_men_shunned
				}
				OR = {
					has_doctrine = doctrine_adultery_women_accepted
					has_doctrine = doctrine_adultery_women_shunned
				}
			}
			add = 50
			if = {
				limit = { has_doctrine = tenet_carnal_exaltation }
				add = 50
			}
		}
	}
}

# Runs in liege context. Available scopes:
#  scope:owner
#  scope:target
#  scope:target_liege
#  scope:capital -- capital barony
# Returns 100 if liege allows, 0 if liege disallows, 50 if undecided
# Simplification: distant relative sex is treated like unrelated sex
court_policy_liege_attitude_to_adultery = {
	if = {
		limit = {
		  scope:owner = { is_close_family_of = scope:target }
		}
		value = court_policy_liege_a2a_related
		if = {
			# if homosexuality is forbidden, return zero
			limit = { is_same_sex_as = yes }
			max = court_policy_liege_a2a_samesex
		}
	}
	else_if = {
		limit = { is_same_sex_as = yes }
		value = court_policy_liege_a2a_samesex
	}
	else = {
		value = court_policy_liege_a2a_general
	}
}

# Called from health.1007 to calculate weight multiplier
court_policy_cancer_value = {
	value = ek_human_age_equivalent
	save_temporary_value_as = my_age
	# default for 40yo : 1
	value = 1
	if = {
		# 0.1 if age < 25
		limit = { scope:my_age < 25 }
		value = 0.1
	}
	else_if = {
		# 0.3 in range 25..35
		limit = { scope:my_age < 35 }
		value = 0.3
	}
	# stays 1 if age < 55, might change if >= 55
	else_if = {
		limit = { scope:my_age >= 55 }
		if = {
			# female, age >= 55: factor of 2
			limit = { is_female = yes }
			value = 2
		}
		else_if = {
			# male, age > 55: factor of 3
			limit = { scope:my_age > 55 }
			value = 3
		}
		# multiply by 2 if >= 65
		if = {
			limit = { scope:my_age >= 65 }
			multiply = 2
		}
		# multiply by 2 if >= 70
		if = {
			limit = { scope:my_age >= 70 }
			multiply = 2
		}
	}
}

# Returns chance that root is willing to contract sanguinare_vampiris. EK
#  calculation: 10 + 40 if ambitious in ek_travel_events.4010. HEAVY.
# Result can theoretically be less than 0 or more than 100.
court_policy_is_willing_to_become_vampire_chance = {
	if = {
		limit = {
			OR = {
				# briarheart, immortal, daedra
				court_policy_can_contract_disease_trigger = no
				has_trait = sanguinare_vampiris
			}
		}
		# zero for immune or already infected
	}
	else = {
		value = 10
		# ai_... personality modifiers
		add = court_policy_greed
		add = court_policy_boldness
		add = court_policy_religion_likes_vampires
		add = court_policy_compassion_for_vampire
		# has rival => need to live longer
		if = {
			limit = { any_relation = { type = rival } }
			add = 10
		}
		if = {
			limit = { any_relation = { type = nemesis } }
			add = 20
		}
		# getting older => more likely
		add = court_policy_age_for_vampire
		# liege is known to be vampire and root likes him
		if = {
			limit = {
				exists = liege
				knows_about_character_being_vampire_trigger = { CHARACTER = liege }
				opinion = {
					target = liege
					value >= high_positive_opinion
				}
			}
			add = 20
		}
		add = court_policy_vampire_friend_or_

		# Additional modifier (such as child that no one else can care about, god
		#  influence, …)
		if = {
			limit = {
				exists = scope:court_policy_desire_to_become_vampire
			}
			add = scope:court_policy_desire_to_become_vampire
		}
	}
}

# Returns number between 0 and 100, more if a relative/friend/lover with good
#  attitude is known as vampire
court_policy_vampire_friend_or_ = {
	if = {
		limit = {
			any_close_family_member = {
				root = {
					opinion = { target = prev value >= high_positive_opinion }
					knows_about_character_being_vampire_trigger = { CHARACTER = prev }
				}
			}
		}
		add = 30
	}
	if = {
		limit = {
			any_relation = {
				type = soulmate
				root = {
					knows_about_character_being_vampire_trigger = { CHARACTER = prev }
				}
			}
		}
		add = 40
	}
	else_if = {
		limit = {
			any_relation = {
				type = lover
				root = {
					knows_about_character_being_vampire_trigger = { CHARACTER = prev }
				}
			}
		}
		add = 20
	}
	if = {
		limit = {
			any_relation = {
				type = best_friend
				root = {
					opinion = { target = prev value >= high_positive_opinion }
					knows_about_character_being_vampire_trigger = { CHARACTER = prev }
				}
			}
		}
		add = 30
	}
	else_if = {
		limit = {
			any_relation = {
				type = friend
				root = {
					opinion = { target = prev value >= high_positive_opinion }
					knows_about_character_being_vampire_trigger = { CHARACTER = prev }
				}
			}
		}
		add = 25
	}
	max = 100
}


# Returns number between -1000 and 100, more if willing to have situational sex
court_policy_spontaneous_sex_desire = {
	if = {
		limit = {
			OR = {
				has_trait = celibate
				is_eunuch_trigger = yes
			}
		add = -1000
	}
	else_if = {
		limit = { has_trait = chaste }
		add = -50
	}
	else = {
		value = stress
		if = {
			limit = { has_trait = lustful }
			add = 50
		}
		if = {
			limit = { has_trait = rakish }
			add = 50
		}
		if = {
			limit = { has_trait = devoted }
			add = {
				value = court_policy_fm_sex
				multiply = 0.33
			}
			# 33 if devoted and religion 100% approves free sex
		}
		if = {
			limit = { has_trait = seducer }
			# Just wanna practice
			add = 40
		}
		if = {
			limit = { is_deviant_trigger = yes }
			add = 40
		}
		max = 100
	}
}


# same as court_policy_spontaneous_sex_desire, but -1000 if can spread disease
court_policy_spontaneous_sex_desire_if_healthy = {
	if = {
		limit = { has_std_trigger = yes }
		add = -1000
	}
	else = {
		add = court_policy_spontaneous_sex_desire
	}
}


# Additive modifier to be applied to court_policy_spontaneous_sex_desire value,
#  based on attitude and health status.
# Available scopes:
#  scope:partner
court_policy_sex_desire_modifier = {
	add = {
		value = "opinion(scope:partner)"
		multiply = 0.33
	}
	if = {
		limit = {
			scope:partner = { has_std_trigger = yes }
		}
		# Disease might spread
		add = -90
	}
}
