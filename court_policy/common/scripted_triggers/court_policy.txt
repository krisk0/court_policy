# yes iff scope:owner and scope:target are of same sex
is_same_sex_as = {
	OR = {
		AND = {
			scope:owner = { is_male = yes }
			scope:target = { is_male = yes }
		}
		AND = {
			scope:owner = { is_female = yes }
			scope:target = { is_female = yes }
		}
	}
}
