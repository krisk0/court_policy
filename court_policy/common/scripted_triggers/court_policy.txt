	OR = {
		AND = {
			scope:owner = { is_male = yes }
			scope:target = { is_male = yes }
		}
		AND = {
			scope:owner = { is_female = yes }
			scope:target = { is_female = yes }
		}
	}
}

# Runs in liege or courtier context. Available scopes:
#  scope:disease_type
court_policy_heal_courtier_or_self = {
	trigger_if = {
		limit = { is_landed = yes }
		court_policy_heal_em = yes
	}
	trigger_else = {
		liege ?= {
			court_policy_heal_em = yes
		}
	}
}


# Runs in liege context. Available scopes:
#  scope:sick_character
#  scope:disease_type
court_policy_heal_em = {
	trigger_if = {
		limit = { scope:disease_type = flag:sanguinare_vampiris }
		court_policy_heal_vampire_uncache = yes
	}
	trigger_else_if = {
		limit = { scope:disease_type = flag:sanies_lupinus }
		court_policy_heal_lycanthrope_uncache = yes
	}
	# For other diseases, no court policy
	trigger_else = { always = yes }
}


court_policy_heal_vampire_uncache = {

		# TODO: how to return value of yes-no variable? Here and in trigger below

		}

}

# Available scope:
#  scope:adultery_spouse
#  scope:sex_partner (optional)
#  scope:unsuccessful_sex_attempt (optional)
# Called either in context of scope:adultery_spouse or scope:sex_partner
#
# court_policy changes: return no if court policy allows the kind of sex that
#  occured
adultery_4001_can_be_imprisoned_trigger = {
	# TODO: check error_log for adultery_4001
	#  adultery_4001_can_be_imprisoned_trigger
	# If no capital province, then cannot imprison
	exists = root.capital_province
	# If error_log is not spammed, then need to drag file events/relations_events/adultery_events.txt
	error_log = "adultery_4001_can_be_imprisoned_trigger overridden"
	# If not my subject, cannot imprison
	root = { allowed_to_imprison_character_trigger = { CHARACTER = prev } }
	if = {
		# If partner is known
		limit = {
			exists = scope:sex_partner
			OR = {
				# … either this is cheating spouse, …
				this = scope:adultery_spouse
				# … or this is sex_partner who really had sex
				NOT = { exists = scope:unsuccessful_sex_attempt }
			}
		}
		save_temporary_scope_value_as = {
			name = att
			value = root.court_policy_liege_attitude_to_adultery_4001
		}
	}
	else = {
		save_temporary_scope_value_as = {
			name = att
			# Assume that was sex with a proper partner
			value = root.court_policy_liege_a2a_general
		}
	}
	# If sex is shunned, should not imprison. Can only imprison if att=0
	scope:att = 0
}

		is_briarheart = no
		#poisoned_blood currently unused, no event sets it
		#has_character_modifier = poisoned_blood
	}
}


# Run in context of a character who potentially will marry. Assume homosexuality
#  is not strictly forbidden. Return yes iff this is boss or boss's court
#  policy approves.
court_policy_allows_same_sex_marriage = {
	OR = {
		# Allow yourself everything
		is_ruler = yes
		# liege must exist, must have chosen his policy, the policy must 100%
		#  approve same sex relation.
		liege.capital_province.var:court_policy_homosexual ?= 100
	}
}

# Returns yes if seduction is forbidden due to being young and sharing blood
# Parameters:
#  $SEDUCER$
#  $TARGET$
guaranteed_under_20_incest_rejection_trigger = {
	# If boss of one of them allows incest, then return no
	NOR = {
		# 50 = there is no strict ban, maybe boss will stop a seduce scheme, maybe not
		# 100 = allowed, boss will never stop a seduce scheme
		$SEDUCER$.liege ?= { court_policy_liege_a2a_related >= 50 }
		$TARGET$.liege ?= { court_policy_liege_a2a_related >= 50 }
	}
	# vanilla. 16yo girl can seduce her twin brother or twin sister. 17yo not
	#  allowed to seduce her 16yo relative
	$TARGET$ = {
		NOT = { is_consort_of = $SEDUCER$ }
		is_close_or_extended_family_of = $SEDUCER$
		age < 20
		age < $SEDUCER$.age
		is_ai = yes #Players are allowed to choose
	}
}
